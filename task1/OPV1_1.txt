#include <iostream>
#include <thread>
#include <chrono>
#include <random>

void printNumbers(const std::string& threadName) {
    std::random_device rd;
    std::mt19937 gen(rd());
    std::uniform_int_distribution<> delay(200, 300);

    for (int i = 1; i <= 5; ++i) {
        std::cout << threadName << ": " << i << std::endl;
        std::this_thread::sleep_for(std::chrono::milliseconds(delay(gen)));
    }
}

int main() {
    std::cout << "Main thread: starting program" << std::endl;

    std::thread t1(printNumbers, "Thread 1");
    std::cout << "Main thread: thread 1 created" << std::endl;

    std::thread t2(printNumbers, "Thread 2");
    std::cout << "Main thread: thread 2 created" << std::endl;

    for (int i = 1; i <= 3; ++i) {
        std::cout << "Main thread: message " << i << std::endl;
        std::this_thread::sleep_for(std::chrono::milliseconds(250));
    }

    std::cout << "Main thread: waiting for threads to complete..." << std::endl;
    t1.join();
    t2.join();

    std::cout << "Main thread: all threads completed, program finished" << std::endl;
    return 0;
}
