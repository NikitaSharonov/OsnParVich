#include <iostream>
#include <complex>
#include <chrono>
#include <omp.h>
#include <opencv2/opencv.hpp>

int main() {
    const int width = 1000;
    const int height = 1000;
    const int max_iter = 1000;
    const double x_min = -2.5, x_max = 1.0;
    const double y_min = -1.5, y_max = 1.5;
    cv::Mat image(height, width, CV_8UC3);

    // последовательная версия 
    auto start_seq = std::chrono::high_resolution_clock::now();
    for (int y = 0; y < height; ++y) {
        for (int x = 0; x < width; ++x) {
            double cx = x_min + (double)x / width * (x_max - x_min);
            double cy = y_min + (double)y / height * (y_max - y_min);
            std::complex<double> c(cx, cy);
            std::complex<double> z(0, 0);
            int iter = 0;
            while (std::abs(z) < 2.0 && iter < max_iter) {
                z = z * z + c;
                ++iter;
            }

            cv::Vec3b color;
            if (iter == max_iter) {
                color = cv::Vec3b(0, 0, 0);
            }
            else {
                int col = (iter % 8) * 32;
                color = cv::Vec3b(col, col, col);
            }
            image.at<cv::Vec3b>(y, x) = color;
        }
    }

    auto end_seq = std::chrono::high_resolution_clock::now();
    std::chrono::duration<double> elapsed_seq = end_seq - start_seq;
    std::cout << "Sequential time: " << elapsed_seq.count() << " sec" << std::endl;
    std::cout << std::endl;

    // параллельные версии для разного числа потоков
    for (int threads = 1; threads <= 4; ++threads) {
        omp_set_num_threads(threads);
        auto start_par = std::chrono::high_resolution_clock::now();

#pragma omp parallel for collapse(2)
        for (int y = 0; y < height; ++y) {
            for (int x = 0; x < width; ++x) {
                double cx = x_min + (double)x / width * (x_max - x_min);
                double cy = y_min + (double)y / height * (y_max - y_min);
                std::complex<double> c(cx, cy);
                std::complex<double> z(0, 0);
                int iter = 0;
                while (std::abs(z) < 2.0 && iter < max_iter) {
                    z = z * z + c;
                    ++iter;
                }

                cv::Vec3b color;
                if (iter == max_iter) {
                    color = cv::Vec3b(0, 0, 0);
                }
                else {
                    int col = (iter % 8) * 32;
                    color = cv::Vec3b(col, col, col);
                }
                image.at<cv::Vec3b>(y, x) = color;
            }
        }

        auto end_par = std::chrono::high_resolution_clock::now();
        std::chrono::duration<double> elapsed_par = end_par - start_par;
        std::cout << "Parallel time (" << threads << " threads): " << elapsed_par.count() << " sec" << std::endl;
        std::cout << "Speedup: " << elapsed_seq.count() / elapsed_par.count() << std::endl;
        std::cout << std::endl;
    }

    cv::imshow("Mandelbrot Set", image);
    cv::waitKey(0);
    return 0;
}
